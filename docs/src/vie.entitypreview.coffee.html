VIE entitypreview uses the VIE.load service method to show a preview for an entity. This demo goes to the default Apache Stanbol backend.

var vie = new VIE();
vie.use(new vie.StanbolService({
    url : "http://dev.iks-project.eu:8081",
    proxyDisabled: true
}));

$('.search')
.entitypreview({
    vie: vie,
    select: function(e, ui){
        console.log(ui);
    }
});

    success = (cacheEntity) =>
      html = ""

      descr = @_getDescription cacheEntity
      unless descr
        @_logger.warn "No description found for", cacheEntity
